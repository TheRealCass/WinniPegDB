SELECT 
    a.Ward AS Ward,
    c.Person AS Councillor,
    c.Phone AS Phone
FROM 
    Addresses a
JOIN 
    Council c
ON 
    a.Ward = c.WardName
WHERE 
    LOWER(a.FullAddress) = LOWER(?)
GROUP BY
	Councillor;

SELECT
    ServiceRequest, Person, ward, year
FROM
    ServiceCalls
JOIN
    Council 
ON
    ServiceCalls.Ward = Council.WardName
GROUP BY
    ServiceRequest;
